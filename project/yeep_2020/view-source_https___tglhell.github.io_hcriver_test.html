<script>
$(function(){
let tarItem = $('.hover-list-outer li');
tarItem.on('mouseenter mouseleave', function(e){
if (!$(this).closest('.hover-list-wrap').hasClass('other-style')) {
let _this = $(this);
let _thisIdx = _this.index();
let itemWid = _this.width();
let contWrap = _this.closest('.hover-list-wrap');
let contItem = contWrap.find('.hover-list-cont .hover-cont-item');
let contItemWid = contItem.eq(_thisIdx).outerWidth();

if (e.type == 'mouseenter') {
_this.addClass('on').siblings().removeClass('on');
contItem.eq(_thisIdx).addClass('active').siblings().removeClass('active');
if (itemWid > contItemWid) {
posItem = -(itemWid - contItemWid) / 2;
} else {
posItem = (contItemWid - itemWid) / 2;
}
contItem.eq(_thisIdx).css('left', (itemWid * _thisIdx) - posItem);
} else {
_this.removeClass('on');
contItem.eq(_thisIdx).removeClass('active');
}
}
});

function itemSize () {
let tarItemWid = tarItem.width();
if ($(window).width() <= 1400) {
$('.hover-list-wrap:not(.other-style)').find('.hover-cont-item').css('max-width', tarItemWid);
} else {
$('.hover-list-wrap:not(.other-style)').find('.hover-cont-item').css('max-width', 'none');
}
}
itemSize ();

$(window).resize(function(){
let tarItemWid = tarItem.width();
itemSize ();
});
});
</script>
